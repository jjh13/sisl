281  virtual lattice_site get_nearest_site(const vector &pt) const {
282  vector xyz = pt * 0.5;
283  vector unit(3), ijk(3), uvw(3), v1, v2;
284 
285  xyz = xyz.array();
286 
287  unit.fill(1.);
288  ijk = xyz + (0.5)*unit;
289  uvw = xyz + unit;
290 
291  for(int i = 0; i < 3; i++) {
292  ijk[i] = 2*floor(ijk[i]);
293  uvw[i] = 2*floor(uvw[i]);
294  }
295  uvw -= unit;
296 
297  v1 = ijk - 2*xyz;
298  v2 = uvw - 2*xyz;
299 
300  if(v1.dot(v1) < v2.dot(v2)) {
301  return ijk.cast<int>();
302  }
303  return uvw.cast<int>();
304  }

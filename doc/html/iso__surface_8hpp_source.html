66  function *f,
67  const double &isoValue,
68  const double &stepSize,
69  sisl::vector origin,
70  sisl::vector extent,
71  bool flip_normal = false,
72  sisl::function *dx = nullptr,

80  if(extent[0] < 0 || extent[1] < 0 || extent[2] < 0) throw "Invalid extent for marching cubes!";
81 
82  sisl::vector boundary = origin + extent;
83  // Figure out the steps in each direction
84  int xres = (int)floor((extent[0])/stepSize);
85  int yres = (int)floor((extent[1])/stepSize);
86  int zres = (int)floor((extent[2])/stepSize);
87  int res = xres*2;
88 
89  map<int, sisl::vertex3 > vMap;
90 
91  #pragma omp parallel for
92  for(int i = 0; i < xres; i++) {
93  float x = origin[0] + i * stepSize;
94  int j = 0;
95 
96  double v000 = 0, v001 = 0, v010 = 0, v100 = 0,
97  v011 = 0, v110 = 0, v101 = 0, v111 = 0;
98  // Break up the inner loops over
99  for(float y = origin[1]; y < boundary[1]; y+=stepSize,j++) {
100  int idx = 0;
101  int k = 0;
102 
103  // Eval these only once
104  v000 = (*f)(x, y, origin[2]);
105  v100 = (*f)(x+stepSize, y, origin[2]);
106  v110 = (*f)(x+stepSize, y+stepSize, origin[2]);
107  v010 = (*f)(x, y+stepSize, origin[2]);
108 
109  idx |= (v000 < isoValue) ? 16 : 0; // {0,0,0}, 0001000
110  idx |= (v100 < isoValue) ? 32 : 0; // {1,0,0}, 0010000
111  idx |= (v110 < isoValue) ? 64 : 0; // {1,1,0}, 0100000
112  idx |= (v010 < isoValue) ? 128 : 0; // {0,1,0}, 1000000
113 
114  for(float z = origin[2]; z < boundary[2]; z+=stepSize, k++) {
115  v001 = (*f)(x, y, z+stepSize);
116  v101 = (*f)(x+stepSize, y, z+stepSize);
117  v111 = (*f)(x+stepSize, y+stepSize, z+stepSize);
118  v011 = (*f)(x, y+stepSize, z+stepSize);
119 
120  idx >>= 4;
121  idx |= (v001 < isoValue) ? 16 : 0 ; // {0,0,1}
122  idx |= (v101 < isoValue) ? 32 : 0 ; // {1,0,1}
123  idx |= (v111 < isoValue) ? 64 : 0 ; // {1,1,1}
124  idx |= (v011 < isoValue) ? 128 : 0 ; // {0,1,1}
125 
126  // Cube is entirely in/out of the surface
127  if (!edges[idx]){
128  v000 = v001;
129  v100 = v101;
130  v110 = v111;
131  v010 = v011;
132  continue;
133  }
134 
135  // Find the vertices where the surface intersects the cube
136  int edgeBit = 1;
137  int vertexIndexList[12];
138 
139  #define MAP_INDEX(x,y,z) (( (((x+i)*res) + (y+j))*res)+(z+k)-1)
140  #define ss stepSize
141  for(int c = 0; c < 12; c++) {
142  if(edges[idx] & edgeBit){
143  switch(c){
144  case 0:
145  vertexIndexList[c] = MAP_INDEX(0,0,0) * 3;//vIndex*3;
146  cache_vertex(&vMap,vertexIndexList[c],
147  vertex3(x,y,z,1,1,1),
148  vertex3(x+ss,y,z,1,1,1),
149  v000,v100,isoValue,ss,f);
150  break;
151 
152  case 2:
153  vertexIndexList[c] = MAP_INDEX(0,1,0) * 3; //(vIndex+zres)*3;
154  cache_vertex(&vMap,vertexIndexList[c],
155  vertex3(x,y+ss,z,1,1,1),
156  vertex3(x+ss,y+ss,z,1,1,1),
157  v010,v110,isoValue,ss,f);
158  break;
159  case 4:
160  vertexIndexList[c] = MAP_INDEX(0,0,1) * 3; //(vIndex+1)*3;
161  cache_vertex(&vMap,vertexIndexList[c],
162  vertex3(x+ss,y,z+ss,1,1,1),
163  vertex3(x,y,z+ss,1,1,1),
164  v101,v001,isoValue,ss,f);
165  break;
166  case 6:
167  vertexIndexList[c] = MAP_INDEX(0,1,1) * 3; //(vIndex+yres+1)*3;
168  cache_vertex(&vMap,vertexIndexList[c],
169  vertex3(x+ss,y+ss,z+ss,1,1,1),
170  vertex3(x,y+ss,z+ss,1,1,1),
171  v111,v011,isoValue,ss,f);
172  break;
174 
175  case 1:
176  vertexIndexList[c] = MAP_INDEX(1,0,0) * 3 + 1;//(vIndex+yres*zres)*3+1;
177  cache_vertex(&vMap,vertexIndexList[c],
178  vertex3(x+ss,y,z,1,1,1),
179  vertex3(x+ss,y+ss,z,1,1,1),
180  v100,v110,isoValue,ss,f);
181  break;
182 
183  case 3:
184  vertexIndexList[c] = MAP_INDEX(0,0,0) * 3 + 1;//vIndex*3+1;
185  cache_vertex(&vMap,vertexIndexList[c],
186  vertex3(x,y,z,1,1,1),
187  vertex3(x,y+ss,z,1,1,1),
188  v000,v010,isoValue,ss,f);
189  break; // 0, 0, 0
190 
191  case 5:
192  vertexIndexList[c] = MAP_INDEX(1,0,1) * 3 + 1;//(vIndex+yres*zres+1)*3+1;
193  cache_vertex(&vMap,vertexIndexList[c],
194  vertex3(x+ss,y,z+ss,1,1,1),
195  vertex3(x+ss,y+ss,z+ss,1,1,1),
196  v101,v111,isoValue,ss,f);
197  break; // 1, 0, 1
198  case 7:
199  vertexIndexList[c] = MAP_INDEX(0,0,1) * 3 + 1;//(vIndex+1)*3+1;
200  cache_vertex(&vMap,vertexIndexList[c],
201  vertex3(x,y,z+ss,1,1,1),
202  vertex3(x,y+ss,z+ss,1,1,1),
203  v001,v011,isoValue,ss,f);
204  break; //0,0,1
205 
207 
208  case 8:
209  vertexIndexList[c] = MAP_INDEX(0,0,0) * 3 + 2;//vIndex*3+2;
210  cache_vertex(&vMap,vertexIndexList[c],
211  vertex3(x,y,z,1,1,1),
212  vertex3(x,y,z+ss,1,1,1),
213  v000,v001,isoValue,ss,f);
214  break; //0,0,0
215  case 9:
216  vertexIndexList[c] = MAP_INDEX(1,0,0) * 3 + 2;//(vIndex+yres*zres)*3+2;
217  cache_vertex(&vMap,vertexIndexList[c],
218  vertex3(x+ss,y,z,1,1,1),
219  vertex3(x+ss,y,z+ss,1,1,1),
220  v100,v101,isoValue,ss,f);
221  break; //1,0,0
222  case 10:
223  vertexIndexList[c] = MAP_INDEX(1,1,0) * 3 + 2;//(vIndex+yres*zres+zres)*3+2;
224  cache_vertex(&vMap,vertexIndexList[c],
225  vertex3(x+ss,y+ss,z,1,1,1),
226  vertex3(x+ss,y+ss,z+ss,1,1,1),
227  v110,v111,isoValue,ss,f);
228  break; //1,1,0
229  case 11:
230  vertexIndexList[c] = MAP_INDEX(0,1,0) * 3 + 2;//(vIndex+yres)*3+2;
231  cache_vertex(&vMap,vertexIndexList[c],
232  vertex3(x,y+ss,z,1,1,1),
233  vertex3(x,y+ss,z+ss,1,1,1),
234  v010,v011,isoValue,ss,f);
235  break; //0,1,0
236  }
237  }
238  edgeBit<<=1;
239  }

244  // Create the triangles
245  for(int c = 0; triangles[idx][c] != -1;){
246 
247  #pragma omp critical (add_face2)
248  {
249  faces.push_back(tuple<int,int,int>(
250  vertexIndexList[triangles[idx][c]],
251  vertexIndexList[triangles[idx][c+1]],
252  vertexIndexList[triangles[idx][c+2]]));
253  c += 3;
254  }
255  }

257  // Setup for next cube
258  v000 = v001;
259  v100 = v101;
260  v110 = v111;
261  v010 = v011;
262  }
263  }
264  }
265  processFaces(&vMap, faces);
266 
267  // Now that we know where all the vertices should lie,
268  // let's figure out the normal directions
269  bool use_grad = false;
270  try {
271  f->grad(0,0,0);
272  use_grad = true;
273  } catch(const char *e) { }
274  bool use_d_func = !(dx == nullptr || dy == nullptr || dz == nullptr);
275 
276  for(unsigned int i = 0; i < plyfile.count_vertices(); i++) {
277  vector x = plyfile.get_vertex_pos(i);
278  vector gv(3);
279 
280  if(use_d_func) {
281  gv << (*dx)(x), (*dy)(x), (*dz)(x);
282  }else if(use_grad) {
283  gv = f->grad(x);
284  }else {
285  double h = m_shadingSs;
286  double ddx = ((*f)(x[0]+h, x[1] , x[2] ) - (*f)(x[0]-h, x[1] , x[2] ))/(2*h);
287  double ddy = ((*f)(x[0], x[1]+h, x[2] ) - (*f)(x[0] , x[1]-h, x[2] ))/(2*h);
288  double ddz = ((*f)(x[0], x[1] , x[2]+h) - (*f)(x[0] , x[1] , x[2]-h))/(2*h);
289  gv << ddx, ddy, ddz;
290  }
291  if(flip_normal)
292  plyfile.set_normal(i, -gv);
293  else
294  plyfile.set_normal(i, gv);
295  }

323  inline void cache_vertex(
324  std::map<int, sisl::vertex3 > *vMap,
325  const int &vIndex,
326  const vertex3 &v1,
327  const vertex3 &v2,
328  const double &iVal1,
329  const double &iVal2,
330  const double &isoValue,
331  const double &ss,
332  sisl::function *f,
333  bool approximate_gradient = false) {
334 
335  #pragma omp critical(cache_vertex)
336  {
337  if(vMap->find(vIndex) == vMap->end()) {
338  double interpolate = (isoValue -iVal1)/(iVal2-iVal1);
339  if(fabs(iVal2-iVal1) < 0.0001)
340  interpolate = 0.5;
341  sisl::vector vnew = v1.p*(1.-interpolate) + v2.p*(interpolate);
342  sisl::vector na(3);// = f->grad(vnew);
343 
344  vMap->insert(
345  {
346  vIndex,
347  sisl::vertex3(vnew, na)
348  });
349  }
350  }

370  int i3 = faceMap[std::get<2>(t)];
371 
372  plyfile.add_triangle(i1,i2,i3);
373  }

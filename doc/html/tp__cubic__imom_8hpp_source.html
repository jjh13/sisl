34  double x = std::abs(t);
35  if(x < 1.) return (1./2.)*x*x*x - x*x - (1./2.)*x + 1;
36  if(x < 2.) return (-1./6.)*x*x*x + x*x - 11./6 *x + 1;

82  lattice_site zero(N), po(N);

88  ret.clear();

91  ret.push_back(l + 2*po);
92  ret.push_back(l + po);

96  }
